#include <stdio.h>#include <stdlib.h>int main(){	float *p=NULL, *old = NULL;	int i=0,j=0;	char answer;		int count = 0, len = 1;//count属入字的个数   len表示开辟空间的大小	float t;		p=(float*)malloc(len*sizeof(float));	while(1)	{		scanf("%f",&t);		count ++;		if(count > len)//如果输入的数比开辟的空间大，则要扩容		{			len *= 2;			old = p;//让old指针指向原数组			p = (float*)malloc(len*sizeof(float));//原数组扩大为2倍						for(i = 0; i < count-1; i++)//将old指向的内容复制给新的原数组				p[i] = old[i];			p[count-1] = t;//保存新输入的数						free(old);//释放old指向的空间			for(j=0;j<count;j++)//测试数据保存是否正确			{				printf("test:%4.2f  ",*(p+j));			}		}		else		{			p[count-1] = t;		}				printf("你是否想继续输入下一个浮点数Y/N:");		getchar();		scanf("%c",&answer);		if(answer=='N')			break;			}		return 0;}
